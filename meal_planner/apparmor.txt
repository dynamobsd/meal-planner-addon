# Profil AppArmor de l'add-on « Planif Repas & Épicerie ».
# Le nom du profil DOIT correspondre au slug de l'add-on (config.yaml : slug: meal_planner).
# Profil minimal et permissif adapté à un backend Python/uvicorn (base python:3.12-slim, sans bashio/s6).

#include <tunables/global>

profile meal_planner flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Réseau (uvicorn écoute sur 8099 ; scraping HTTP sortant)
  network inet stream,
  network inet6 stream,
  network netlink raw,

  # Autoriser l'envoi de signaux d'arrêt aux processus fils
  signal (send) set=(kill,term,int,hup,cont),

  # Exécution de l'interpréteur Python et des binaires système du conteneur
  /usr/local/bin/python3* ix,
  /usr/local/bin/python* ix,
  /usr/bin/python3* ix,
  /usr/bin/python* ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/local/bin/** ix,

  # Script de démarrage de l'add-on
  /run.sh rix,

  # Code et environnement de l'application (lecture seule + import de modules)
  /app/** r,
  /usr/local/lib/python3*/** mr,
  /usr/lib/python3*/** mr,

  # Fichiers temporaires
  /tmp/ rw,
  /tmp/** rwk,

  # Données persistantes de l'add-on : base SQLite, options.json, etc. (lecture/écriture)
  /data/ rw,
  /data/** rwk,
}
